--Touch Event--

local touchPart = game.Workspace.TouchPart

partIsTouched = false

touchPart.Touched:Connect(function(otherPart)
	if partIsTouched == false then
		partIsTouched = true 
		print(otherPart.Name)
		task.wait(1)
		partIsTouched = false
	end
	
	if  touchPart == partIsTouched then
		touchPart.Color= Color3(163,162,40)
	end
	
end)
